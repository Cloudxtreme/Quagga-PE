#if defined __STDC_VERSION__ && __STDC_VERSION__ >= 199901L

#define zfpm_debug(...)                                         \
  do {                                                          \
    if (IS_ZEBRA_DEBUG_FPM) zlog_debug("FPM: " __VA_ARGS__);    \
  } while(0)

#elif defined __GNUC__

  do {                                                  \
    if (IS_ZEBRA_DEBUG_FPM) zlog_debug("FPM: " _args);  \
  } while(0)
