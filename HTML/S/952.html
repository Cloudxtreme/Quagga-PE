#include <zebra.h>

int
ipforward ()
{
  int fd, ret;
  int af = AF_INET;
  char netopt[] = "ipforwarding";
  struct optreq oq;

  fd = socket(af, SOCK_DGRAM, 0);
  if (fd < 0) {
    /* need logging here */
    return -1;
  }

  strcpy (oq.name, netopt);
  oq.getnext = 0;

  ret = ioctl (fd, SIOCGNETOPT, (caddr_t)&oq);
  close(fd);

  if (ret < 0) {
    /* need logging here */
    return -1;
  }

  ret = atoi (oq.data);
  return ret;
}
